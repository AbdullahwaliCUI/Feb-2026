<script>
function openJob(evt, jobName) {
    var i, tabcontent, tablinks;
    tabcontent = document.getElementsByClassName("tab-content");
    for (i = 0; i < tabcontent.length; i++) {
        tabcontent[i].classList.remove("active");
    }
    tablinks = document.getElementsByClassName("job-tab");
    for (i = 0; i < tablinks.length; i++) {
        tablinks[i].classList.remove("active");
    }
    document.getElementById(jobName).classList.add("active");
    evt.currentTarget.classList.add("active");
    
    // Auto-select in dropdown if possible
    var select = document.getElementById("jobTitle");
    var jobTitle = evt.currentTarget.textContent;
    for (var j = 0; j < select.options.length; j++) {
        if (select.options[j].value === jobTitle) {
            select.selectedIndex = j;
            updateSubject();
            break;
        }
    }
}

function updateSubject() {
    var select = document.getElementById("jobTitle");
    var subject = document.getElementById("formSubject");
    if (select.value) {
        subject.value = "New Job Application - " + select.value;
    }
}
</script>
